>>> def fib(n):    #用迭代算斐波那契
	n1 = 1
	n2 = 1
	n3 = 1
	if n == 2 or n == 1:
		return 1
	while (n-2)>0:
		n3 = n2 + n1
		n1 = n2
		n2 = n3
		n = n-1
	return n3
